:: OrdIkone.bat
:: Ikone einer Datei dem zugehöringen Ordner zuordnen
:: Aufruf durch das "Senden an ..." Menü im Explorer
:: nur für die Verwendung unter 4DOS
:: Version 0.1 - 01.Juni 2001 - Stephan Frye
:: Version 0.2 - 28.Juni 2001 - Stephan Frye

@echo off

::echo Eingabe ist: %1

::echo Datei ist: %@filename[%1]
::echo Pfad ist: %@path[%1]

echo [.ShellClassInfo]        >>%@path[%1]desktop.ini
echo IconFile=%@filename[%1]  >>%@path[%1]desktop.ini
echo IcoIndex=0               >>%@path[%1]desktop.ini
echo InfoTip=                 >>%@path[%1]desktop.ini

::type %@path[%1]desktop.ini

ATTRIB +h %@path[%1]desktop.ini
ATTRIB /d +r %@path[%1]

:Ende
