Exact Audio Copy Einstellungen für die Kompression mit externem lame.exe

EAC 0.99

-V 2 --vbr-new --id3v2-only --pad-id3v2 --ignore-tag-errors --tt "%t" --ta "%a" --tl "%g" --ty "%y" --tn "%n" --tg "%m" %s %d

EAC 1

-V 2 --vbr-new --id3v2-only --pad-id3v2 --ignore-tag-errors --tt "%title%" --ta "%artist%" --tl "%albumtitle%" --ty %year% --tn "%tracknr%/%numtracks%" --tg "%genre%" "%source%" "%dest%"
