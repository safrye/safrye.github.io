@echo off
:: repeat_finder starten mit Angabe der Xmere und der FASTA-Sequenz
:: Syntax: rfinder.bat [FASTA-Datei] [Zahl(Min-Xmer)] [Zahl(Max-Xmer)]

title Repeat-finder Batch
::Variablen setzen
set _arbeitsweg_=c:\Dienst\bin
set msgbox=%_arbeitsweg_%\msgbox.exe
if not exist %msgbox% goto kein_msgbox
set wprompt=%_arbeitsweg_%\wprompt.exe
set winput=%_arbeitsweg_%\winput.exe
::set rfinder=D:\Data\Perl\NRC_Institute_for_Biological_Sciences\repeat_finder\repeat_finder.pl
set rfinder=C:\Dienst\perl\repeat_finder\repeat_finder.pl

if "%1"=="" goto datei-fehler

%msgbox% "The FASTA sequence file must be clean!" "Warning" 2 2 8
if %errorlevel% == 2 goto Ende

set weg=%~dp1
set min=%2
set max=%3
:min
if "%min%"=="" call :min-eingabe
if "%min%"=="" goto min
:max
if "%max%"=="" call :max-eingabe
if "%max%"=="" goto max
echo FASTA-Datei: %1
echo Min-Xmer: %min%
echo Max-Xmer: %max%
start "MsbBox" /b %msgbox% "Min=%min% bp Max=%max% bp Repeats in der Datei %1" "Eingabe" 1 1

for /L %%x in (%min%,+1,%max%) do call :rf-start %%x %%1 %weg%
goto Ende


:Ende
goto:eof

:: Unterprogramme
:rf-start
echo.
echo --------------------------------------------------
echo %1-mer:
%rfinder% -n%1 "%2" >> "%3%~n2.rfXmer"
goto:eof

:min-eingabe
%winput% "set min=$input" "Minimale Repeatlaenge (bp)" /num > %tmp%\temp.bat
if not errorlevel 1 call %tmp%\temp.bat
goto:eof

:max-eingabe
%winput% "set max=$input" "Maximale Repeatlaenge (bp)" /num > %tmp%\temp.bat
if not errorlevel 1 call %tmp%\temp.bat
goto:eof

:datei-fehler
echo Dateiname fehlt!
echo Syntax: rfinder.bat [FASTA-Datei] [Zahl(Min-Xmer)] [Zahl(Max-Xmer)]
start "MsbBox" /b %msgbox% "Syntax: rfinder.bat [FASTA-Datei] [Zahl(Min-Xmer)] [Zahl(Max-Xmer)]" "Dateiname fehlt" 2 1
goto Ende
